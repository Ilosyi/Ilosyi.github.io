字符串与字符相关函数的使用

2023-11-19 C语言学习笔记（万老师） 大一

1.sscanf,sprintf,atoi函数

例题1：P1957 口算练习题 - 洛谷 | 计算机科学教育新生态 (luogu.com.cn)

scanf和atoi解法

    #include<stdio.h>
    #include <cctype>
    #include<string.h>

    int main() {
	int x, p, q;
	scanf("%d",&x);
	for (int i = 0; i <x; i++) 
	{
		char m='a';
		char s[20] = { 0 }, d[30] = { 0 };
		scanf("%s",s);   //scanf读取字符串时读取到空格即停止
		if (isalpha(s[0])) {
			m = s[0];
			scanf("%d%d", &p,&q);
		}
		else {
			p = atoi(s);
			scanf("%d",&q);
		}
		if (m == 'a')
			sprintf(d,"%d+%d=%d", p, q, p + q);
		else if (m == 'b')
			sprintf(d, "%d-%d=%d", p, q,p - q);
		else
			sprintf(d, "%d*%d=%d", p, q, p * q);
		printf("%s\n%d\n", d,strlen(d));
	}
	return 0;
    }

gets和sscanf解法

    #include<stdio.h>
    #include<string.h>
    int main()
    {
	int x;
	scanf("%d ",&x);
	char m;
	for (int l = 0; l < x; l++)
	{
		char a[20] = { 0 };
		char b[10] = { 0 };
		char c[10] = { 0 };
		char d[30] = { 0 };
		int p, q; int n = 2;
		gets(a);
		if (a[0] >= 'a' && a[0] <= 'c')
		{
			m = a[0];		n = 0;
		}
		int i; int u = 0;
		for (i = 2 - n;a[i] != ' ';i++)
		{
			b[u] = a[i];
			u++;
		}
		int j = i;
		for (i = j + 1; a[i] != '\0'; i++)
		{
			c[i - j - 1] = a[i];
		}
		sscanf(b, "%d", &p);
		sscanf(c, "%d", &q);
		if (m == 'a')
			sprintf(d, "%d+%d=%d", p, q, p + q);
		else if (m == 'b')
			sprintf(d, "%d-%d=%d", p, q, p - q);
		else
			sprintf(d, "%d*%d=%d", p, q, p * q);
		printf("%s\n%d\n", d, strlen(d));
	}
	return 0;
    }
